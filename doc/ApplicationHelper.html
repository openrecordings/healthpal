def days_ago(date_time)
  days = (DateTime.now.to_date - date_time.to_date).to_i
  case days
  when 0
    t(:today)
  when 1
    t(:yesterday)
  else
    t(:days_ago).gsub('#', days.to_s)
  end
end

def mm_ss(seconds, end_seconds = 0)
  mm = (seconds / 60).floor
  ss = "%02d" % (seconds % 60).to_i
  if end_seconds == 0
    "#{mm}:#{ss}"
  else
    "#{mm}:#{ss} - #{mm_ss(end_seconds)}"
  end
end
